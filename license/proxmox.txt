# Create the VM

qm create 202 \
  --name cbresponse2 \
  --memory 32768 \
  --cores 4 \
  --net0 virtio,bridge=vmbr0 \
  --scsihw virtio-scsi-pci \
  --ostype l26 \
  --cpu host \
  --serial0 socket \
  --agent 1 \
  --boot order=scsi0

# Import QCOW2 disk

qm importdisk 202 /var/lib/vz/images/202/Rocky-8-GenericCloud-Base.latest.x86_64.qcow2 local-lvm

# Attach disk and cloud-init config

qm set 202 --scsi0 local-lvm:vm-202-disk-0
qm set 202 --ide2 local:cloudinit
qm set 202 --boot order=scsi0
qm set 202 --cicustom "user=local:snippets/user-data.yaml"
qm set 202 --ide2 local-lvm:cloudinit
qm cloudinit update 202
qm config 202

# Set Static IP 
# qm set 201 --nameserver 1.1.1.1,8.8.8.8

qm set 202 --ipconfig0 ip=192.168.1.30/24,gw=192.168.1.99
qm set 202 --nameserver 198.142.152.164
qm cloudinit update 202


# Start VM and verify agent ping

qm start 202
qm agent 202 ping
qm reboot 202

## Uninstall 

# Stop Carbon Black EDR services
echo "Stopping Carbon Black EDR services..."
sudo /usr/share/cb/cbcluster stop
sudo service cb-enterprise stop 2>/dev/null

# Remove Carbon Black EDR packages
echo "Removing Carbon Black EDR packages..."
sudo yum remove -y cb-enterprise cb-datagrid cb-datastore cb-solr cb-swagger cbui libselinux-cb-python python-cb-coreservices python-cb-response-venv

# Delete remaining directories
echo "Deleting remaining directories..."
sudo rm -rf /var/www/cb /var/run/cb /var/log/cb /var/lib/cb /var/cb /usr/share/cb /etc/cb

# Clean up orphaned dependencies
echo "Cleaning up orphaned dependencies..."
sudo yum autoremove -y

# Remove Sensor
/opt/carbonblack/response/bin/sensoruninstall.sh


# Reboot server
echo "Uninstall complete. Rebooting server..."
